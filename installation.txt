	ORBSLAM2 DEPENDENCIES:
		C++11 or C++0x Compiler

		opencv 3.0: no need if installing ros kinetic
		https://docs.opencv.org/master/d7/d9f/tutorial_linux_install.html

		pangolin:
		https://github.com/stevenlovegrove/Pangolin

		eigen3:
		sudo apt install libeigen3-dev

		BLAS and LAPACK:
		sudo apt-get install libblas-dev
		sudo apt-get install liblapack-dev


	ROS: kinetic
	http://wiki.ros.org/ROS/Installation
	sudo apt-get install ros-kinetic-image-common
	sudo apt-get install ros-kinetic-image-pipeline

	PYTHON: can use conda
	sudo apt install python-pip
	exit
	pip install numpy
	pip install scipy
	pip install sklearn

BUILD:
mkdir formula_ws
cd formuls_ws
mkdir src
cd src
git clone https://github.com/tomnorman/cone_slam.git
git clone https://github.com/yardenshap/custom_msgs.git
cd orb_slam2
./build.sh
catkin build
echo 'source ~/formula_ws/devel/setup.bash' >> ~/.bashrc
exit
